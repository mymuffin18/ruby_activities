CREATE TABLE students (id SERIAL PRIMARY KEY NOT NULL,
					   first_name varchar(255) NOT NULL,
					   middle_name varchar(255) DEFAULT 'Blank',
					   last_name varchar(255) NOT NULL,
					   age integer NOT NULL,
					   location varchar(255) NOT NULL
					  );
